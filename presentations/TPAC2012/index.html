The <picture> Element

<picture width="500" height="500">
     <source media="(min-width: 45em)" 
                src="large.jpg">
   
     <source media="(min-width: 18em)" 
                src="med.jpg">
   
     <source src="small.jpg">
   
     <img src="small.jpg" alt="">
   
     <p>Accessible text</p> 
</picture>
